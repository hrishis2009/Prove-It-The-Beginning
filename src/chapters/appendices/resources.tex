\chapter{Additional Resources}

\section{Modular Arithmetic}
\begin{itemize}
    \item A nice AOPS Introduction to Modular Arithmetic \url{https://artofproblemsolving.com/wiki/index.php/Modular_arithmetic/Introduction}
    \item A nice handout for Modular Arithmetic: \url{https://artofproblemsolving.com/articles/files/SatoNT.pdf}
    \item Another  nice handout for Modular Arithmetic: \url{https://circles.math.ucla.edu/circles/lib/data/Handout-867-959.pdf}
    \item Yet another nice handout for Modular Arithmetic: \url{https://davidaltizio.web.illinois.edu/ModularArithmetic.pdf}
    \item A good book if you want to go deeper into number theory (especially in the context of competition mathematics): \url{https://artofproblemsolving.com/store/book/intro-number-theory}
    \item Modern Olympiad Number Theory - A (free) book that is basically zero-hero in Number theory in the context of Math Olympiads: \url{https://drive.google.com/file/d/1BcJTLjQaelZ4w_70oHKyImC2I8zLfyrt/view}
\end{itemize}

\section{Competition Mathematics}
\begin{itemize}
    \item A platform for all things math (especially competition math). Check out the forums, handouts, previous competitions, Alcumus, etc. : \url{https://artofproblemsolving.com/}
    \item A link for problems and solutions for MAA-affiliated mathematics competitions (AMC, AIME, USA(J)MO): \url{https://artofproblemsolving.com/wiki/index.php/AMC_Problems_and_Solutions}
    \item Art of Problem Solving Volume 1: The Basics by Sandor Lehoczky and Richard Rusczyk (VERY helpful even if you are not a beginner): \url{https://artofproblemsolving.com/store/book/aops-vol1} 
    \item Art of Problem Solving Volume 2: and Beyond by Sandor Lehoczky and Richard Rusczyk: \url{https://artofproblemsolving.com/store/book/aops-vol2}
    \item A list of books and resources organized by topic: \url{https://artofproblemsolving.com/wiki/index.php/Olympiad_books}
    \item A list of more resources for math competitions: \url{https://artofproblemsolving.com/wiki/index.php?title=Resources_for_mathematics_competitions}
    \item Evan Chen's website for Olympiad resources: \url{https://web.evanchen.cc/olympiad.html}
    \item A (my) YouTube channel with competition math problems explained. \url{https://youtube.com/@continuummath}
\end{itemize}
